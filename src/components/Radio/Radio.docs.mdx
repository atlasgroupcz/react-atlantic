# Radio

Component to display switch between defined options

## Common usage

```tsx
import React from 'react';
import { RadioButtons, RadioButtonProps } from '@atlasgroup/react-atlantic';

const Component = () => {
    const values = [
        { value: 'UK', label: 'Anglie ðŸ‡¬ðŸ‡§' },
        { value: 'CZ', label: 'ÄŒesko ðŸ‡¨ðŸ‡¿' },
    ];
    const [value, setValue] = useState<string>(values[0]);

    const onChange: RadioButtonProps['onChange'] = (e) => setValue(e.currentTarget.value);

    return <RadioButtons values={values} value={value} onChange={onChange} groupName="groupToFocusBy" />;
};
```
