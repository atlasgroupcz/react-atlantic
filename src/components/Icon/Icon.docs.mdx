import { Playground } from 'docz';

import { IconsMap } from '.';

export const supportedNames = Object.keys(IconsMap);

# Icon

Icon component provides set of basic svg icons commonly used in web apps

## Supported Icons

Name of specific icon is in title of each icon - just hover icon to see its name

<Playground>
    <WrapperHorizontal>
        {supportedNames.map((name) => (
            <Icon name={name} title={name} key={name} />
        ))}
    </WrapperHorizontal>
</Playground>

## Rotating prop

Make css animation of rotating icon... use wisely, propably only for `loading` icon

<Playground>
    <Icon name={'loading'} isRotating />
    <Icon name={'edit'} isRotating />
</Playground>
